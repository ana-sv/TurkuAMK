    def takePicture():
    cap = cv2.VideoCapture(0) # video capture source camera (Here webcam of laptop) 
    ret,frame = cap.read() # return a single frame in variable `frame`
    cap.release()
    return frame

    def displayPicture(frame):
        cv2.imshow('image',frame) #display the captured image
        cv2.waitKey(0)
        cv2.destroyAllWindows()

    Step 2: Capture an image.¶

    The code below captures an image and displays it. First, any existing camera is closed to ensure we are starting from the beginning with the camera initialization.     image = takePicture()
cv2.imwrite("test.jpg", image) 

pltFrame = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
# The plt.show() method is used as if the image were a graph.
# Turn off the axis
plt.axis('off')
# Title of the picture
plt.title("Webcam Video/Image")
# show the frame
plt.imshow(pltFrame)
plt.show()
